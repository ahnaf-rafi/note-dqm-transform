\section{Proofs of results in \Cref{sec--dqm-def}}
% \label{sec--prf--intro}

\subsection{Proof of \Cref{lem--hellinger-diff-properties}}
\label{sec--prf--lem--hellinger-diff-properties}

\subsubsection{Proof of part \ref{lem--hellinger-diff-properties-q-diff-in-L1}
of \Cref{lem--hellinger-diff-properties}}

From the expression for \(\rho_{\theta, \theta_{0}}\) in
\eqref{eqn--diff-def-remainder},
\begin{align*}
  q_{\theta} - q_{\theta_{0}} - (\theta - \theta_{0})^{\prime}
  \dot{q}_{\theta_{0}} =
  & \,
  g_{\theta}^{2} - g_{\theta_{0}}^{2} - (\theta - \theta_{0})^{\prime} \left( 2
  g_{\theta_{0}} \dot{g}_{\theta_{0}} \right) \\
  =
  & \, \left( g_{\theta_{0}} + (\theta - \theta_{0})^{\prime}
  \dot{g}_{\theta_{0}} + \rho_{\theta, \theta_{0}} \right)^{2} -
  g_{\theta_{0}}^{2} - 2 g_{\theta_{0}} \cdot (\theta - \theta_{0})^{\prime}
  \dot{g}_{\theta_{0}} \\
  =
  & \, \left( (\theta - \theta_{0})^{\prime} \dot{g}_{\theta_{0}} \right)^{2} +
  \rho_{\theta, \theta_{0}}^{2} + 2 g_{\theta_{0}} \rho_{\theta, \theta_{0}} + 2
  (\theta - \theta_{0})^{\prime} \dot{g}_{\theta_{0}} \rho_{\theta, \theta_{0}}.
\end{align*}
By the triangle inequality and the Cauchy-Schwarz inequality,
\begin{equation*}
  \left| q_{\theta} - q_{\theta_{0}} - (\theta - \theta_{0})^{\prime}
  \dot{q}_{\theta_{0}} \right| \leq |\theta - \theta_{0}|^{2} \left|
  \dot{g}_{\theta_{0}} \right|^{2} + \rho_{\theta, \theta_{0}}^{2} + 2
  g_{\theta_{0}} \left| \rho_{\theta, \theta_{0}} \right| + 2 |\theta -
  \theta_{0}| \left| \dot{g}_{\theta_{0}} \right| \left| \rho_{\theta,
  \theta_{0}} \right|.
\end{equation*}
Hence,
\begin{equation*}
  \begin{split}
    \mu \left[ \left| q_{\theta} - q_{\theta_{0}} - (\theta -
    \theta_{0})^{\prime} \dot{q}_{\theta_{0}} \right| \right] \leq
    & \, |\theta - \theta_{0}|^{2} \mu \left[ \left| \dot{g}_{\theta_{0}}
    \right|^{2} \right] + \mu \left[ \rho_{\theta, \theta_{0}}^{2} \right] \\
    & + 2 \mu \left[ g_{\theta_{0}} \left| \rho_{\theta, \theta_{0}} \right|
    \right] + 2 |\theta - \theta_{0}| \mu \left[ \left| \dot{g}_{\theta_{0}}
    \right| \left| \rho_{\theta, \theta_{0}} \right| \right].
  \end{split}
\end{equation*}
By the Cauchy-Schwarz inequality (this time in \(\mathscr{L}_{2} (\mu)\)),
\begin{align*}
  \mu \left[ \left| q_{\theta} - q_{\theta_{0}} - (\theta - \theta_{0})^{\prime}
  \dot{q}_{\theta_{0}} \right| \right] \leq
  & \, |\theta - \theta_{0}|^{2} \mu \left[ \left| \dot{g}_{\theta_{0}}
  \right|^{2} \right] + \mu \left[ \rho_{\theta, \theta_{0}}^{2} \right] + 2
  \sqrt{\mu \left[ g_{\theta_{0}}^{2} \right] \mu \left[ \left| \rho_{\theta,
  \theta_{0}} \right|^{2} \right]} \\
  & + 2 |\theta - \theta_{0}| \sqrt{\mu \left[ \left| \dot{g}_{\theta_{0}}
  \right|^{2} \right] \mu \left[ \left| \rho_{\theta, \theta_{0}} \right|^{2}
  \right]} \\
  =
  & \, O \left( |\theta - \theta_{0}|^{2} \right) + o \left( |\theta -
  \theta_{0}|^{2} \right) + o \left( |\theta - \theta_{0}| \right) + |\theta -
  \theta_{0}| o (|\theta - \theta_{0}|) \\
  =
  & \, o (|\theta - \theta_{0}|).
\end{align*}
\qed

\subsubsection{Proof of part
\ref{lem--hellinger-diff-properties-q-singularity-result} of
\Cref{lem--hellinger-diff-properties}}

Use the remainder \(\rho_{\theta, \theta_{0}}\) defined in
\eqref{eqn--diff-def-remainder} to write \(g_{\theta} = g_{\theta_{0}} + (\theta
- \theta_{0})^{\prime} \dot{g}_{\theta_{0}} + \rho_{\theta, \theta_{0}}\).
On the set \(\left\{ g_{\theta_{0}} = 0 \right\}\),
\(g_{\theta} = (\theta - \theta_{0})^{\prime} \dot{g}_{\theta_{0}} +
\rho_{\theta, \theta_{0}}\).
Hence,
\begin{equation*}
  \begin{split}
    \mu \left[ g_{\theta}^{2} \mathbf{1} \left\{ g_{\theta_{0}} = 0 \right\}
    \right] =
    & \, \mu \left[ \left( (\theta - \theta_{0})^{\prime} \dot{g}_{\theta_{0}} +
    \rho_{\theta, \theta_{0}} \right)^{2} \mathbf{1} \left\{ g_{\theta_{0}} = 0
    \right\} \right]
    \leq
    2 |\theta - \theta_{0}|^{2} \mu \left[ \left|
    \dot{g}_{\theta_{0}} \right|^{2} \right] + 2 \mu \left[ \rho_{\theta,
    \theta_{0}}^{2} \right] \\
    =
    & \, O \left( |\theta - \theta_{0}|^{2} \right) + o \left( |\theta -
    \theta_{0}|^{2} \right) = O \left( |\theta - \theta_{0}|^{2} \right),
  \end{split}
\end{equation*}
which proves \eqref{eqn--hellinger-diff-properties-q-singularity-result-1}.

Now we prove \eqref{eqn--hellinger-diff-properties-q-singularity-result}.
Let \(\tau = \theta - \theta_{0}\).
By assumption \(\theta_{0} + \tau = \theta \in \Theta\) and since \(\theta_{0}
\in \mathrm{int} (\Theta)\), for \(|\tau|\) sufficiently small we have
\(\theta_{0} - \tau \in \Theta\) as well.
Then
\begin{equation*}
  g_{\theta_{0} + \tau} + g_{\theta_{0} - \tau} = \left( \tau^{\prime}
  \dot{g}_{\theta_{0}} + \rho_{\theta_{0} + \tau, \theta_{0}} \right) + \left( -
  \tau^{\prime} \dot{g}_{\theta_{0}} + \rho_{\theta_{0} - \tau, \theta_{0}}
  \right) = \rho_{\theta_{0} + \tau, \theta_{0}} + \rho_{\theta_{0} - \tau,
  \theta_{0}}.
\end{equation*}
By non-negativity, \(0 \leq g_{\theta} \leq g_{\theta_{0} + \tau} +
g_{\theta_{0} - \tau}\) and so by the inequality \((u + v)^{2} \leq 2 \left(
u^{2} + v^{2} \right)\), \(g_{\theta}^{2} \leq \left( g_{\theta_{0} + \tau} +
g_{\theta_{0} - \tau} \right)^{2} \leq 2 \rho_{\theta_{0} + \tau,
\theta_{0}}^{2} + 2 \rho_{\theta_{0} - \tau, \theta_{0}}^{2}\).
Therefore by \eqref{eqn--diff-def-remainder},
\begin{equation*}
  \mu \left[ g_{\theta}^{2} \cdot \mathbf{1} \left\{ g_{\theta_{0}} = 0 \right\}
  \right] \leq 2 \mu \left[ \rho_{\theta_{0} + \tau, \theta_{0}}^{2} \right] + 2
  \mu \left[ \rho_{\theta_{0} - \tau, \theta_{0}}^{2} \right] = o \left(
  |\tau|^{2} \right) = o \left( |\theta - \theta_{0}|^{2} \right),
\end{equation*}
which establishes \eqref{eqn--hellinger-diff-properties-q-singularity-result}.
\qed

\subsubsection{Proof of part
\ref{lem--hellinger-diff-properties-dotg-singularity-result}
of \Cref{lem--hellinger-diff-properties}}

Use \eqref{eqn--diff-def-remainder} to write
\((\theta - \theta_{0})^{\prime} \dot{g}_{\theta_{0}} = g_{\theta} -
g_{\theta_{0}} - \rho_{\theta, \theta_{0}}\).
On \(\left\{ g_{\theta_{0}} = 0 \right\}\), \((\theta - \theta_{0})^{\prime}
\dot{g}_{\theta_{0}} = g_{\theta} - \rho_{\theta, \theta_{0}}\).
Hence \(\left( (\theta - \theta_{0})^{\prime} \dot{g}_{\theta_{0}} \right)^{2}
\leq 2 g_{\theta}^{2} + 2 \rho_{\theta, \theta_{0}}^{2}\).
Thus, by \eqref{eqn--diff-def-remainder} and
\eqref{eqn--hellinger-diff-properties-q-singularity-result} in part
\ref{lem--hellinger-diff-properties-q-singularity-result} (now proven),
\begin{equation*}
  \mu \left[ \left( (\theta - \theta_{0})^{\prime} \dot{g}_{\theta_{0}}
  \right)^{2} \mathbf{1} \left\{ g_{\theta_{0}} = 0 \right\} \right] \leq 2 \mu
  \left[ g_{\theta}^{2} \mathbf{1} \left\{ g_{\theta_{0}} = 0 \right\} \right] +
  2 \mu \left[ \rho_{\theta, \theta_{0}}^{2} \mathbf{1} \left\{ g_{\theta_{0}} =
  0 \right\} \right] = o \left( |\theta - \theta_{0}|^{2} \right).
\end{equation*}
Then we get that with \(C = \mu \left[ \dot{g}_{\theta_{0}}
\dot{g}_{\theta_{0}}^{\prime} \mathbf{1} \left\{ g_{\theta_{0}} = 0 \right\}
\right]\)
\begin{equation*}
  (\theta - \theta_{0})^{\prime} C (\theta - \theta_{0}) \leq \psi \left(
  |\theta - \theta_{0}|^{2} \right),
\end{equation*}
for some function \(\psi (\cdot)\) that satisfies \(\psi (w) / |w|^{2} \to 0\)
as \(|w| \to 0\).
We are done if we show that \(C\) is the zero matrix, or equivalently that
\(v^{\prime} C v\) for any \(v \neq \mathbf{0}\).
First note that \(C\) is positive semi-definite and so \(v^{\prime} C v \geq
0\).
Set \(h = \alpha v\) for \(\alpha > 0\).
Then
\begin{equation*}
  \alpha^{2} v^{\prime} C v \leq \psi(\alpha v), \text{ and so }
  v^{\prime} C v \leq \frac{\psi(\alpha v)}{|\alpha v|^{2}} \cdot |v|^{2}.
\end{equation*}
Since \(|v| \neq 0\) and \(\psi(h) = o(|\tau|^{2})\) as \(|\tau| \to 0\), taking
\(\alpha \to 0\) sends the right side to zero.
So \(v^{\prime}Cv \leq 0\) for every \(v\).
Combined with \(v^{\prime} C v \geq 0\), we get \(v^{\prime} C v = 0\).
Since \(v\) was arbitrary, \(C\) is the zero matrix, which establishes
\eqref{eqn--hellinger-diff-properties-gdot-singularity-result}.
\qed
